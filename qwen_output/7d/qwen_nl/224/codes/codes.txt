                    event_in_which_city = {}
                    for i, city_id in enumerate(city_ids):
                        if i < len(transportation_times) - 1:
                            if transportation_times[i+1] <= transportation_times[i]:
                                event_in_which_city[i] = city_ids[i]

                    event_index = {}

                def get_event_length(events, event_index):
                    event_lengths = {}

                        event_lengths[key] = len(events[events.keys()[key]]) - index
                    
                    return event_lengths
                
                def maximize_music_events(events, event_in_which_city, event_index, category_scores):
                    max_score = 0
                    best_selection = []
                    for i in range(len(event_in_which_city)):
                        if event_in_which_city[i] == event_in_which_city[i-1] and i > 0:
                            continue
                        current_score = 0
                        for j in range(i, len(event_in_which_city)):
                            if event_in_which_city[j] != event_in_which_city[i]:
                                break
                            current_score += category_scores["Music-related events"] * event_lengths[j]
                        
                        if current_score > max_score:
                            max_score = current_score
                            best_selection = [i for i in range(i, j+1)]
                    
                    return best_selection
