
    # Calculate taxi price for 1 person and 2 transportations based on driving distance
    def calculate_taxi_price(distance, base_fare=3.5, per_mile_rate=2.0):
        total_price = base_fare + (distance * per_mile_rate)
        return total_price * 2  # For 2 transportations
    
    # Example usage
    distance = 10  # Example distance in miles
    price = calculate_taxi_price(distance)
    print(f"Total taxi price for 2 transportations: ${price}")
    
    # Get taxi arrival time with Chicago as origin and final destination, specific city, and departure date for 2 transportations
    from datetime import datetime, timedelta
    
    def get_taxi_arrival_time(origin, destination, departure_date, travel_time_hours=1):
        # Simulate travel time calculation
        departure_datetime = datetime.strptime(departure_date, "%Y-%m-%d %H:%M")
        arrival_datetime = departure_datetime + timedelta(hours=travel_time_hours)
        return arrival_datetime
    
    # Example usage
    origin = "Chicago"
    destination = "Specific City"
    departure_date = "2023-10-15 08:00"
    arrival_time = get_taxi_arrival_time(origin, destination, departure_date)
    print(f"Estimated arrival time: {arrival_time.strftime('%Y-%m-%d %H:%M')}")
